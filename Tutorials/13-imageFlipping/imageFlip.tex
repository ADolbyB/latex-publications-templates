% Vertical flip
\begin{figure}
   \begin{subfigure}[b]{.5\textwidth}
     \begin{tikzpicture}
         \node[scale=0.4] (a)
         { \scalebox{1}[1]{\includegraphics[scale=0.2]{2019m8.jpg}} };
     \end{tikzpicture}
     \caption{Original Image}
   \end{subfigure}
   \begin{subfigure}[b]{.5\textwidth}
     \begin{tikzpicture}
         \node[scale=0.4] (a)
         { \scalebox{1}[-1]{\includegraphics[scale=0.2]{2019m8.jpg}} };  % Note: \scalebox{1}[-1] (Flipped Y-axis)
      \end{tikzpicture}
      \caption{Flipped Image}
   \end{subfigure}
\caption{Vertical flip of .jpg or .pdf files}
\end{figure}
